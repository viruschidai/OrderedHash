fetch

() Boolean

Defined in orderedhash.js:101

Fetch an element in the hash and do something else if the element does not exist fetch(key[, default]) 

 var hash =  new OrderedHash();
 hash.set('key1', 'value1');
 var value = hash.fetch('key2', 'value2');
 console.log(value); // the value will be 'value2' although it does not exist in the hash

fetch(key, callback(key))

 var hash =  new OrderedHash();
 hash.set('key1', 'value1');
 var value = hash.fetch('key2', function(key){
      return 'value2';
 });
 console.log(value); // the value will be 'value2' although it does not exist in the hash

Returns:

Boolean: Returns true when there is no element in the hash
